# 백준 4577번 소코반

## 문제

![https://www.acmicpc.net/problem/4577](https://www.acmicpc.net/problem/4577)


## 코드

```c++
#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <vector>
#include <cstring>
#define LEN 18
using namespace std;
int R, C, goal = 0;
char map[LEN][LEN];
pair<int, int> player;
void Print(int g)
{
	cout << "Game " << g << ": " << ((goal == 0) ? "complete" : "incomplete") << endl;

	for (int i = 0; i < R; i++) {
		cout << map[i] << endl;
	}
}
bool isout(char c)
{
	return (c == 'b' || c == 'B' || c == '#');
}
void move(int y, int x)
{
	int cury = player.first, curx = player.second;
	char& cur = map[cury][curx];
	char& next = map[cury + y][curx + x];

	// 벽이면 리턴
	if (next == '#') return;

	if (next == 'b' || next == 'B') {

		char& neig = map[cury + (y * 2)][curx + (x * 2)];
		if (isout(neig)) return;		// 박스옆이 박스거나 벽이면 리턴

		if (neig == '+') {
			neig = 'B';
			goal--;
		}
		else {
			neig = 'b';
		}

		if (next == 'B') {
			next = '+';
			goal++;
		}
		else {
			next = '.';
		}
	}
	cur = cur == 'W' ? '+' : '.';
	next = next == '+' ? 'W' : 'w';
	player = { cury + y, curx + x };
}
int main(void)
{
	int game = 0;
	while (1) {

		cin >> R >> C;
		if (R == 0) break;
		memset(map, 0, sizeof(char) * LEN * LEN);
		goal = 0;

		for (int i = 0; i < R; i++) {
			cin >> map[i];
		}

		for (int i = 0; i < R; i++) {
			for (int j = 0; j < C; j++) {
				if (map[i][j] == 'w' || map[i][j] == 'W') {
					player.first = i;
					player.second = j;
				}

				if (map[i][j] == 'b')
					goal++;
			}
		}

		string oper;
		cin >> oper;
		int ny, nx;

		for (char& dir : oper) {

			if (goal == 0) break;
			switch (dir) {
			case 'U':
				ny = -1; nx = 0; break;
			case 'D':
				ny = 1; nx = 0; break;
			case 'L':
				ny = 0; nx = -1; break;
			case 'R':
				ny = 0; nx = 1; break;
			default:break;
			}
			move(ny, nx);
			if (!goal) break;
		}
		Print(++game);
	}
	return 0;
}
```
