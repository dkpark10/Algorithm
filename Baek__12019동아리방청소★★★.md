# 백준 12019번 동아리방 청소

[https://www.acmicpc.net/problem/12019](https://www.acmicpc.net/problem/12019}

## 코드 

```c++

#include <string>
#include <cstring> 
#include <queue>
#include <iostream>
#include <map>
#include <algorithm>
#include <vector>
#include <cstdlib>
#include <random>
#include <cstdio>
#include <ctime>
#include <sstream>
#include <set>

#pragma warning(disable:4996)

#define MAX(a,b) ((a) > (b) ? (a) : (b))
#define MIN(a,b) ((a) > (b) ? (b) : (a))

#define INF1 1e15
#define INF2 987654321
#define INF3 1e9
#define LL long long
#define VS vector<string>
#define VI vector<int>
#define VLL vector<LL>
#define VB vector<bool>
#define VC vector<char> 
#define PII pair<int,int>
#define PLL pair<LL,LL>
#define DIVI 1000000

using namespace std;
int N, M, people[110], dp[110][11][110];
VI ans;

int go(int today, int cleaning, int last, int dirty) {
	
	if (today > N) {
		return 0;
	}

	int& ref = dp[today][cleaning][last];
	if (ref != -1) {
		return ref;
	}

	// 청소 안한다.
	ref = go(today + 1, cleaning, last, dirty + people[today]);
	// 청소한다.
	if (cleaning > 0) {
		ref = MIN(ref, go(today + 1, cleaning - 1, today, 0));
	}

	return ref += people[today] * dirty;
}

void trace(int today, int cleaning, int last) {
	
	if (cleaning <= 0) {
		return;
	}

	if (dp[today + 1][cleaning - 1][today] <= dp[today + 1][cleaning][last]) {
		ans.push_back(today);
		trace(today + 1, cleaning - 1, today);
	}
	else {
		trace(today + 1, cleaning, last);
	}
}

int main(void) {

	scanf("%d %d", &N, &M);
	for (int i = 1; i <= N; i++) {
		scanf("%d", &people[i]);
	}

	memset(dp, -1, sizeof(dp));
	printf("%d\n", go(1, M, 0, 0));
	trace(1, M, 0);
	for (int& day : ans) {
		printf("%d ", day);
	}

	return 0;
}
```
