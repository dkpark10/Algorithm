# 백준 2110번 공유기 설치

## 문제 

![https://www.acmicpc.net/problem/2110](https://www.acmicpc.net/problem/2110)

## 코드

```c++
#include <string>
#include <cstring> 
#include <queue>
#include <iostream>
#include <map>
#include <algorithm>
#include <vector>
#include <cstdlib>
#include <random>
#include <cstdio>
#include <ctime>
#include <stack>
#include <sstream>
#include <set>

#pragma warning(disable:4996)

#define MAX(a,b) ((a) > (b) ? (a) : (b))
#define MIN(a,b) ((a) > (b) ? (b) : (a))

#define INF1 1e15
#define INF2 987654321
#define INF3 1e9
#define LL long long
#define ULL unsigned long long
#define VS vector<string>
#define VI vector<int>
#define VLL vector<LL>
#define VB vector<bool>
#define VC vector<char> 
#define PII pair<int,int>
#define PLL pair<LL,LL>
#define DIVI 1000000000
using namespace std;

int n, c;
VI home;
int main(void) {
	
	scanf("%d %d", &n, &c);
	for (int i = 0; i < n; i++) {
		int val;
		scanf("%d", &val);
		home.push_back(val);
	}
	sort(home.begin(), home.end());

	int l = 1, r = home[n - 1];
	while (l <= r) {
		int mid = (l + r) / 2;
		int cnt = 1;
		int prev = home[0];
		for (int i = 1; i < n; i++) {
			if (home[i] - prev >= mid) {
				prev = home[i];
				cnt++;
			}
		}
		if (cnt >= c) {
			l = mid + 1;
		}
		else {
			r = mid - 1;
		}
	}
	printf("%d\n", r);
	return 0;
}
```
