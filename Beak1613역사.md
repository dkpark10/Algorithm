```c++
#include <string>
#include <algorithm>
#include <vector>
#include <cstdlib>
#include <iostream>
#include <queue>

using namespace std;

#define PII pair<int,int>
#define VI vector<int>
#define VB vector<bool>
#define VLL vector<long long>
#define ll long long
#define INF 987654321

vector<VI>nexthis, prevhis;
bool go(int cur, int desti, VB& chk) {

	bool ret = false;
	if (desti == cur) {
		return true;
	}

	chk[cur] = true;
	for (int i = 0; i < nexthis[cur].size(); i++) {
		int next = nexthis[cur][i];
		if (chk[next] == false) {
			ret = go(next, desti, chk);
			if (ret) return ret;
		}
	}
	return ret;
}

int main(void)
{
	int N, K, a, b;
	scanf("%d %d", &N, &K);
	nexthis.resize(N + 1);
	prevhis.resize(N + 1);

	for (int i = 0; i < K; i++) {
		scanf("%d %d", &a, &b);
		nexthis[a].push_back(b);
		prevhis[b].push_back(a);
	}

	int S;
	scanf("%d", &S);

	VB nextchk(N + 1);
	VB prevchk(N + 1);

	for (int i = 0; i < S; i++) {

		scanf("%d %d", &a, &b);
		fill(nextchk.begin(), nextchk.end(), false);
		fill(prevchk.begin(), prevchk.end(), false);
		
		bool res1 = go(a, b, nextchk);
		bool res2 = go(b, a, prevchk);

		if (res1) {
			printf("%d\n", -1);
		}
		else if (res2) {
			printf("%d\n", 1);
		}
		else {
			printf("%d\n", 0);
		}
	}

	return 0;
}
```

## 풀이 

이전역사 다음역사를 그래프로 기록 도착지에 도달하는지 돌아주면 된다.
